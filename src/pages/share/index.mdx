export function isPptFile(url = '') {
  return /\.(ppt|pptx)(\?.*)?$/i.test(url);
}

export function isVideoFile(url = '') {
  return /\.(mp4|webm|ogg|mov|m4v)(\?.*)?$/i.test(url);
}

export function isPdfFile(url = '') {
  return /\.(pdf)(\?.*)?$/i.test(url);
}

export function isPrivateHostname(hostname = '') {
  const host = String(hostname).toLowerCase();
  if (!host) return true;
  if (host === 'localhost' || host === '127.0.0.1' || host === '0.0.0.0' || host.endsWith('.local')) return true;
  if (/^10\./.test(host)) return true;
  if (/^192\.168\./.test(host)) return true;
  if (/^172\.(1[6-9]|2\d|3[0-1])\./.test(host)) return true;
  return false;
}

export function getAbsoluteResourceUrl(rawUrl = '', siteBaseUrl = '') {
  if (!rawUrl) return '';
  if (/^https?:\/\//i.test(rawUrl)) return rawUrl;

  const runtimeBase = typeof window === 'undefined' ? '' : window.location.origin;
  const base = runtimeBase || siteBaseUrl;
  if (!base) return '';

  try {
    return new URL(rawUrl, base).toString();
  } catch (e) {
    return '';
  }
}

export function getOfficePreviewUrl(rawUrl = '', siteBaseUrl = '') {
  if (!rawUrl) return '';
  const absoluteUrl = getAbsoluteResourceUrl(rawUrl, siteBaseUrl);
  if (!absoluteUrl) return '';

  try {
    const parsed = new URL(absoluteUrl);
    if (isPrivateHostname(parsed.hostname)) return '';
  } catch (e) {
    return '';
  }

  let decodedAbsoluteUrl = absoluteUrl;
  try {
    decodedAbsoluteUrl = decodeURI(absoluteUrl);
  } catch (e) {
    decodedAbsoluteUrl = absoluteUrl;
  }

  return `https://view.officeapps.live.com/op/embed.aspx?src=${encodeURIComponent(decodedAbsoluteUrl)}`;
}

export function UIIcon({name, className = ''}) {
  const commonProps = {
    className,
    viewBox: '0 0 24 24',
    width: '1em',
    height: '1em',
    'aria-hidden': 'true',
    focusable: 'false',
    fill: 'none',
    stroke: 'currentColor',
    strokeWidth: '2',
    strokeLinecap: 'round',
    strokeLinejoin: 'round',
  };

  if (name === 'chevron-left') {
    return (
      <svg {...commonProps}>
        <polyline points="15 18 9 12 15 6" />
      </svg>
    );
  }
  if (name === 'chevron-right') {
    return (
      <svg {...commonProps}>
        <polyline points="9 18 15 12 9 6" />
      </svg>
    );
  }
  if (name === 'meeting') {
    return (
      <svg {...commonProps}>
        <rect x="3" y="6" width="13" height="12" rx="2" />
        <polygon points="16 10 21 8 21 16 16 14" />
      </svg>
    );
  }
  if (name === 'slides') {
    return (
      <svg {...commonProps}>
        <rect x="4" y="4" width="16" height="12" rx="2" />
        <line x1="9" y1="20" x2="15" y2="20" />
        <line x1="12" y1="16" x2="12" y2="20" />
      </svg>
    );
  }
  if (name === 'video') {
    return (
      <svg {...commonProps}>
        <polygon points="10 8 16 12 10 16 10 8" />
        <circle cx="12" cy="12" r="9" />
      </svg>
    );
  }
  if (name === 'drive') {
    return (
      <svg {...commonProps}>
        <path d="M7 17 3.5 11 7 5h10l3.5 6-3.5 6H7z" />
        <line x1="3.5" y1="11" x2="20.5" y2="11" />
      </svg>
    );
  }
  if (name === 'eye') {
    return (
      <svg {...commonProps}>
        <path d="M2 12s3.8-6 10-6 10 6 10 6-3.8 6-10 6-10-6-10-6z" />
        <circle cx="12" cy="12" r="2.5" />
      </svg>
    );
  }
  if (name === 'download') {
    return (
      <svg {...commonProps}>
        <path d="M12 3v11" />
        <polyline points="8 11 12 15 16 11" />
        <path d="M4 19h16" />
      </svg>
    );
  }
  if (name === 'link') {
    return (
      <svg {...commonProps}>
        <path d="M10 14 14 10" />
        <path d="M8 17H6a4 4 0 0 1 0-8h2" />
        <path d="M16 7h2a4 4 0 1 1 0 8h-2" />
      </svg>
    );
  }
  if (name === 'copy') {
    return (
      <svg {...commonProps}>
        <rect x="9" y="9" width="10" height="10" rx="2" />
        <rect x="5" y="5" width="10" height="10" rx="2" />
      </svg>
    );
  }
  if (name === 'check') {
    return (
      <svg {...commonProps}>
        <polyline points="5 13 10 17 19 8" />
      </svg>
    );
  }

  return (
    <svg {...commonProps}>
      <circle cx="12" cy="12" r="9" />
    </svg>
  );
}

export function SharePage() {
  const {siteConfig} = useDocusaurusContext();
  const baseUrl = siteConfig?.baseUrl || '/';
  const siteBaseUrl = useMemo(() => {
    const siteUrl = siteConfig?.url || '';
    if (!siteUrl) return '';
    try {
      return new URL(baseUrl, siteUrl).toString();
    } catch (e) {
      return siteUrl;
    }
  }, [siteConfig?.url, baseUrl]);

  const shares = Array.isArray(shareData?.shares) ? shareData.shares : [];
  const shareById = useMemo(() => new Map(shares.map((item) => [item.id, item])), [shares]);

  const defaultId = shares[0]?.id || '';
  const [activeId, setActiveId] = useState(defaultId);
  const activeShare = shareById.get(activeId);

  function resolvePreviewPayload(item, fallbackType = 'slides', fallbackTitle = '资料预览') {
    const sourceType = item?.type || fallbackType;
    const primaryUrl = item?.downloadUrl || item?.url || '';
    const isVideo = sourceType === 'video' || isVideoFile(primaryUrl);
    const absolutePrimaryUrl = getAbsoluteResourceUrl(primaryUrl, siteBaseUrl);

    if (isVideo && primaryUrl) {
      return {type: 'video', url: primaryUrl, title: item?.name || fallbackTitle, absoluteUrl: absolutePrimaryUrl};
    }

    if (item?.previewUrl) {
      return {type: 'slides', url: item.previewUrl, title: item?.name || fallbackTitle, absoluteUrl: getAbsoluteResourceUrl(item.previewUrl, siteBaseUrl)};
    }

    if (isPptFile(primaryUrl)) {
      return {
        type: 'slides',
        url: getOfficePreviewUrl(primaryUrl, siteBaseUrl),
        title: item?.name || fallbackTitle,
        absoluteUrl: absolutePrimaryUrl,
      };
    }

    if (isPdfFile(primaryUrl)) {
      return {type: 'slides', url: primaryUrl, title: item?.name || fallbackTitle, absoluteUrl: absolutePrimaryUrl};
    }

    return {type: '', url: '', title: item?.name || fallbackTitle, absoluteUrl: absolutePrimaryUrl};
  }

  const previewableResources = useMemo(() => {
    if (!activeShare?.resources?.length) return [];
    const entries = [];

    activeShare.resources.forEach((resource) => {
      const topPreview = resolvePreviewPayload(resource, resource.type || 'slides', resource.name || '资料预览');
      if (topPreview.url) entries.push(topPreview);

      const files = Array.isArray(resource.files) ? resource.files : [];
      files.forEach((file, index) => {
        const fileTitle = file?.name || `${resource.name || '资料'} 文件${index + 1}`;
        const filePreview = resolvePreviewPayload(file, resource.type || 'slides', `${resource.name || '资料'} - ${fileTitle}`);
        if (filePreview.url) entries.push({...filePreview, title: `${resource.name || '资料'} - ${fileTitle}`});
      });
    });

    return entries;
  }, [activeShare, siteBaseUrl]);

  const [activePreview, setActivePreview] = useState({type: '', url: '', title: ''});
  const [copiedKey, setCopiedKey] = useState('');
  const [toast, setToast] = useState({visible: false, text: ''});
  const [isSidebarCollapsed, setIsSidebarCollapsed] = useState(false);

  useEffect(() => {
    if (typeof window === 'undefined') return;
    const hash = decodeURIComponent(window.location.hash || '').replace(/^#/, '').trim();
    if (hash && shareById.has(hash)) setActiveId(hash);
  }, [shareById]);

  useEffect(() => {
    if (typeof window === 'undefined') return;
    if (!activeId) return;
    window.history.replaceState(null, '', `#${encodeURIComponent(activeId)}`);
  }, [activeId]);

  useEffect(() => {
    if (!previewableResources.length) {
      setActivePreview({type: '', url: '', title: ''});
      return;
    }
    const first = previewableResources[0];
    const firstVideoUrl = first.url || first.downloadUrl || '';
    const previewUrl =
      first.type === 'video'
        ? firstVideoUrl
        : first.previewUrl || (isPptFile(first.downloadUrl) ? getOfficePreviewUrl(first.downloadUrl) : first.downloadUrl);
    setActivePreview({
      type: first.type || '',
      url: previewUrl || '',
      title: first.name || '资料预览',
    });
  }, [previewableResources]);

  async function handleCopyPassword(password, key) {
    if (!password || typeof window === 'undefined') return;
    try {
      if (navigator?.clipboard?.writeText) {
        await navigator.clipboard.writeText(password);
      } else {
        const textarea = document.createElement('textarea');
        textarea.value = password;
        textarea.setAttribute('readonly', '');
        textarea.style.position = 'fixed';
        textarea.style.opacity = '0';
        document.body.appendChild(textarea);
        textarea.select();
        document.execCommand('copy');
        document.body.removeChild(textarea);
      }
      setCopiedKey(key);
      setToast({visible: true, text: '访问密码已复制到剪贴板'});
      window.setTimeout(() => {
        setCopiedKey((current) => (current === key ? '' : current));
      }, 1600);
      window.setTimeout(() => {
        setToast((current) => (current.visible ? {...current, visible: false} : current));
      }, 1800);
    } catch (e) {
      setToast({visible: true, text: '复制失败，请手动复制'});
      window.setTimeout(() => {
        setToast((current) => (current.visible ? {...current, visible: false} : current));
      }, 1800);
    }
  }

    return '打开资源链接';
  }

    if (type === 'slides') return 'PPT资料';
    if (type === 'video') return '视频';
    if (type === 'buaa-drive') return '北航网盘';
    return '资源';

    if (type === 'meeting') return 'meeting';
    if (type === 'slides') return 'slides';
    if (type === 'video') return 'video';
    if (type === 'buaa-drive') return 'drive';
    return 'link';

              aria-label={isSidebarCollapsed ? '展开左侧标签栏' : '折叠左侧标签栏'}
            >

              <span className="shareSidebarToggleText">{isSidebarCollapsed ? '展开' : '折叠'}</span>
            </button>

            <div className="shareTimeline">
              {shares.map((item, itemIndex) => {

                return (
                  <button
                    key={item.id}
                    onClick={() => setActiveId(item.id)}
                    className={`shareTimelineItem ${isActive ? 'isActive' : ''}`}
                    style={{animationDelay: `${itemIndex * 70}ms`}}
                    type="button"
                  >
                    <div className="shareItemLine">{item.date} [{item.tag}]</div>
                    <div className="shareItemTitle">{item.title}</div>
                  </button>
                );
              })}
            </div>
          </div>
        </aside>

        <section className="shareContent" key={activeId}>
        <div className="shareStageHeader">
          <div style={{fontWeight: 650}}>{activeShare?.title || '学业分享'}</div>
          <div style={{opacity: 0.6, fontSize: '0.9rem'}}>{activeShare?.date || ''}</div>
        </div>

        <div className="shareStageBody">
          {!activeShare ? (
            <div style={{padding: 18, opacity: 0.75}}>暂无分享内容。</div>
          ) : (
            <div style={{padding: 18, display: 'grid', gap: 14}}>
              {activeShare.summary ? (
                <div style={{opacity: 0.8}}>{activeShare.summary}</div>
              ) : null}

              <div style={{display: 'grid', gap: 12}}>
                {(activeShare.resources || []).map((resource, index) => {
                  const resourceKey = `${resource.name}-${index}`;
                  const resourceFiles = Array.isArray(resource.files) ? resource.files : [];
                  const hasFiles = resourceFiles.length > 0;
                  const hasDownload = !!resource.downloadUrl;
                  const previewData = resolvePreviewPayload(resource, resource.type || 'slides', resource.name || '资料预览');
                  const previewUrl = previewData.url;
                  const isVideoResource = previewData.type === 'video';
                  const hasPreview = !!previewUrl;
                  const hasPassword = !!resource.password;
                  const hasOpenLink = !!resource.url;
                  const shouldShowPending = !hasDownload && !hasOpenLink && !hasPreview && !hasFiles;

                  return (
                    <div key={resourceKey} className="shareResourceCard" style={{animationDelay: `${index * 55}ms`}}>
                      <div className="shareResourceInfo">
                        <div className="shareResourceTitleRow">
                          <span className="shareTypeIcon" aria-hidden="true">
                            <UIIcon name={getResourceIconName(resource.type)} />
                          </span>
                          <div className="shareResourceTitle">{resource.name}</div>
                          <span className="shareResourceType">{getResourceTypeLabel(resource.type)}</span>
                        </div>
                        {hasPassword ? (
                          <div className="shareResourceMeta">
                            {resource.type === 'buaa-drive' ? '提取码' : '访问密码'}：{resource.password}
                          </div>
                        ) : null}
                        {resource.note ? <div className="shareResourceNote">{resource.note}</div> : null}
                      </div>
                      <div className="shareActionRow">
                        {hasPreview ? (
                          <button
                            type="button"
                            className="button button--secondary button--sm shareActionButton"
                            onClick={() =>
                              setActivePreview({
                                type: previewData.type || (isVideoResource ? 'video' : 'slides'),
                                url: previewUrl,
                                title: resource.name || '资料预览',
                              })
                            }
                          >
                            <span className="shareActionIcon" aria-hidden="true"><UIIcon name="eye" /></span>
                            在线预览
                          </button>
                        ) : null}
                        {hasDownload ? (
                          <a
                            className="button button--secondary button--sm shareActionButton"
                            href={resource.downloadUrl}
                            target="_blank"
                            rel="noreferrer"
                            download
                          >
                            <span className="shareActionIcon" aria-hidden="true"><UIIcon name="download" /></span>
                            下载文件
                          </a>
                        ) : null}
                        {shouldShowPending ? (
                          <span style={{fontSize: '0.88rem', opacity: 0.6}}>资源待上传</span>

                          <a

                            rel="noreferrer"
                          >
                            <span className="shareActionIcon" aria-hidden="true"><UIIcon name="link" /></span>

                      {hasFiles ? (

                          {resourceFiles.map((file, fileIndex) => {
                            const fileKey = `${resourceKey}-file-${fileIndex}`;
                            const fileData = resolvePreviewPayload(file, resource.type || 'slides', file?.name || `文件${fileIndex + 1}`);
                            const hasFilePreview = !!fileData.url;
                            const hasFileDownload = !!file?.downloadUrl;
                            const hasFileOpenLink = !!file?.url;

                                        setActivePreview({
                                          type: fileData.type || 'slides',
                                          url: fileData.url,
                                          title: `${resource.name || '资料'} - ${file?.name || `文件${fileIndex + 1}`}`,
                                        })
                                      }
                                    >
                                      <span className="shareActionIcon" aria-hidden="true"><UIIcon name="eye" /></span>
                                      在线预览
                                    </button>
                                  ) : null}
                                  {hasFileDownload ? (
                                    <a
                                      className="button button--secondary button--sm shareActionButton"
                                      href={file.downloadUrl}
                                      target="_blank"
                                      rel="noreferrer"
                                      download
                                    >
                                      <span className="shareActionIcon" aria-hidden="true"><UIIcon name="download" /></span>
                                      下载文件
                                    </a>
                                  ) : null}
                                  {hasFileOpenLink ? (
                                    <a
                                      className="button button--secondary button--sm shareActionButton"
                                      href={file.url}
                                      target="_blank"
                                      rel="noreferrer"
                                    >
                                      <span className="shareActionIcon" aria-hidden="true"><UIIcon name="link" /></span>
                                      {getOpenLinkLabel(file)}
                                    </a>

                                  {hasFilePassword ? (
                                    <button
                                      type="button"
                                      className={`button button--sm shareActionButton ${copiedKey === fileKey ? 'button--success' : 'button--secondary'}`}
                                      onClick={() => handleCopyPassword(file.password, fileKey)}
                                    >
                                      <span className="shareActionIcon" aria-hidden="true">
                                        <UIIcon name={copiedKey === fileKey ? 'check' : 'copy'} />
                                      </span>
                                      {copiedKey === fileKey ? '已复制' : '复制密码'}
                                    </button>

                                </div>
                              </div>
                            );
                          })}

                      ) : null}
                    </div>
                  );
                })}
              </div>

              {activePreview.url ? (
                <div className="sharePreviewWrap" key={activePreview.url}>
                  <div className="sharePreviewTitle">{activePreview.title} 在线预览</div>
                  {activePreview.type === 'video' ? (
                    <video className="sharePreviewFrame" controls preload="metadata">

                    </video>
                  ) : (
                    <iframe
                      title="分享资料预览"
                      src={activePreview.url}
                      className="sharePreviewFrame"
                    />
                  )}
                </div>
              ) : null}

          )}

      </div>
